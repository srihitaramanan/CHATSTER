CREATE TABLE `user` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`first_name` VARCHAR(255) NULL DEFAULT NULL,
	`last_name` VARCHAR(255) NULL DEFAULT NULL,
	`department` VARCHAR(255) NULL DEFAULT NULL,
	`year` VARCHAR(255) NULL DEFAULT NULL,
	`group_name` VARCHAR(2000) NULL DEFAULT NULL,
	`email_address` VARCHAR(255) NOT NULL,
	`password` VARCHAR(255) NOT NULL,
	`is_active` SMALLINT(6) NOT NULL DEFAULT '1',
	`row_created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	`row_altered` TIMESTAMP NULL DEFAULT NULL,
	PRIMARY KEY (`id`)
)

CREATE TABLE `role` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`authority` VARCHAR(255) NULL DEFAULT NULL,
	`is_active` SMALLINT(6) NULL DEFAULT NULL,
	`row_altered` DATETIME NULL DEFAULT NULL,
	`row_created` DATETIME NULL DEFAULT NULL,
	PRIMARY KEY (`id`)
)

CREATE TABLE `user_role` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`is_active` SMALLINT(6) NULL DEFAULT NULL,
	`row_altered` DATETIME NULL DEFAULT NULL,
	`row_created` DATETIME NULL DEFAULT NULL,
	`role_fk` INT(11) NULL DEFAULT NULL,
	`user_fk` INT(11) NULL DEFAULT NULL,
	PRIMARY KEY (`id`),
	INDEX `FK143BF46AB057D52E` (`role_fk`),
	INDEX `FK_user_role_user` (`user_fk`),
	CONSTRAINT `FK_user_role_role` FOREIGN KEY (`role_fk`) REFERENCES `role` (`id`) ON UPDATE NO ACTION ON DELETE NO ACTION,
	CONSTRAINT `FK_user_role_user` FOREIGN KEY (`user_fk`) REFERENCES `user` (`id`) ON UPDATE NO ACTION ON DELETE NO ACTION
)
CREATE TABLE `college_group` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`group_name` VARCHAR(250) NULL DEFAULT NULL,
	`is_active` SMALLINT(6) NULL DEFAULT '1',
	`row_created` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
	`row_altered` TIMESTAMP NULL DEFAULT NULL,
	INDEX `Index 1` (`id`)
)


CREATE TABLE `department` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`department_name` VARCHAR(255) NULL DEFAULT NULL,
	`is_active` SMALLINT(6) NULL DEFAULT NULL,
	`row_altered` DATETIME NULL DEFAULT NULL,
	`row_created` DATETIME NULL DEFAULT NULL,
	PRIMARY KEY (`id`)
)

CREATE TABLE `year` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`is_active` SMALLINT(6) NULL DEFAULT NULL,
	`row_altered` DATETIME NULL DEFAULT NULL,
	`row_created` DATETIME NULL DEFAULT NULL,
	`year` VARCHAR(255) NULL DEFAULT NULL,
	PRIMARY KEY (`id`)
)
